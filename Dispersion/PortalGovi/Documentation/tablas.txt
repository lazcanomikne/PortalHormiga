CREATE COLUMN TABLE MIKNE.COTIZACION_ENCABEZADO (
    ID                INTEGER GENERATED BY DEFAULT AS IDENTITY NOT NULL,
    TIPO_COTIZACION   NVARCHAR(30),
    TIPO_CUENTA       NVARCHAR(30),
    IDIOMA_COTIZACION NVARCHAR(30),
    CLIENTE           NVARCHAR(100),
    CONTACTO          NVARCHAR(100),
    DIR_FISCAL        NVARCHAR(200),
    DIR_ENTREGA       NVARCHAR(200),
    REFERENCIA        NVARCHAR(200),
    TERMINOS_ENTREGA  NVARCHAR(200),
    FOLIO_PORTAL      INTEGER,
    FOLIO_SAP         INTEGER,
    FECHA             DATE,
    VENCIMIENTO       DATE,
    MONEDA            NVARCHAR(10),
    PRIMARY KEY (ID)
);


CREATE COLUMN TABLE "MIKNE"."COTIZACION_PRODUCTOS" (
    ID                INTEGER GENERATED BY DEFAULT AS IDENTITY NOT NULL,
    ID_COTIZACION     INTEGER,         -- Relación a la cotización principal
    NUM_ARTICULO      NVARCHAR(50),
    CANTIDAD          DECIMAL(18,2),
    PRECIO_VENTA      DECIMAL(18,2),
    ASIGNADO_A        NVARCHAR(100),
    DEFINICIONES      NVARCHAR(500),
    PRIMARY KEY (ID),
    FOREIGN KEY (ID_COTIZACION) REFERENCES "MIKNE"."COTIZACION_ENCABEZADO" (ID)
);

CREATE COLUMN TABLE "MIKNE"."COTIZACION_BAHIAS" (
    ID                 INTEGER GENERATED BY DEFAULT AS IDENTITY NOT NULL,
    ID_COTIZACION      INTEGER,        -- Relación a la cotización principal
    BAHIA              NVARCHAR(100),
    ALIMENTACION       NVARCHAR(100),
    RIEL               NVARCHAR(100),
    ESTRUCTURA         NVARCHAR(100),
    DEFINICIONES       NVARCHAR(500),
    PRIMARY KEY (ID),
    FOREIGN KEY (ID_COTIZACION) REFERENCES "MIKNE"."COTIZACION_ENCABEZADO" (ID)
);
